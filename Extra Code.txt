// Drop all non-country data.
drop if strpos(RecipientName, "regional") != 0
drop if strpos(RecipientName, "unspecified") != 0
drop if strpos(RecipientName, "Yugoslavia") != 0

// reshapes
// Aid Move data into a wide format
reshape wide Value_USD, i(Year RecipientName) j(SDG_num)

// Remove periods from indicator codes so they can be used as varnames
replace indicatorcode = subinstr(indicatorcode, ".", "_", .)

// Put WB Data in wide format
reshape wide yr*, i(countryname) j(indicatorcode) string

// UNStat Data in wide format
reshape wide Value, i(GeoAreaName TimePeriod) j(SeriesCode) string
// Don't use the below
reshape wide Value*, i(GeoAreaName) j(TimePeriod)
